
var preguntas = ['Cuél es tu nombre?',
                 'Cuantos años tienes?',
                 'Lenguaje de programacion favorito?'];

var respuestas = [ ];

function pregunta(i){
     process.stdout.write(preguntas[i]);
}

process.stdin.on('data', function(data){
   respuestas.push(data.toString().trim());
    if(respuestas.length < preguntas.length){
        pregunta(respuestas.length);
    }else{
     process.exit();
}
});

process.on('exit', function() {
    process.stdout.write(`el nombre es ${respuestas[0]} tiene ${respuestas[1]} años, y su lenguaje de programacion favorita es  ${respuestas[2]}` ); 
})
pregunta(0);
